/* ==========================================================================
   DDL (Data Definition Language): Used to define the database structure.
   ==========================================================================
*/

-- 1. Defining the Student Table
CREATE TABLE student (
    -- [Key Constraint]: PRIMARY KEY (defines entity integrity)
    -- [Domain Constraint]: UNSIGNED (only positive numbers) and MEDIUMINT (limit 16.7M)
    roll_no MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT, 
    
    -- [Domain Constraint]: VARCHAR(20) limits name length
    first_name VARCHAR(20), 
    
    -- [Domain Constraint]: DATE format (YYYY-MM-DD)
    dob DATE, 
    branch VARCHAR(10),
    
    PRIMARY KEY(roll_no)
) ENGINE = InnoDB AUTO_INCREMENT = 2403001; 
-- ENGINE = InnoDB provides [Referential Integrity] support.


-- 2. Defining the Book Table
CREATE TABLE book (
    -- [Domain Constraint]: Fixed 13 chars, ASCII charset for memory efficiency
    isbn CHAR(13) CHARACTER SET ascii NOT NULL, 
    title VARCHAR(50),
    author VARCHAR(30),
    publisher VARCHAR(50),
    
    -- [Key Constraint]: PRIMARY KEY
    PRIMARY KEY(isbn)
);


-- 3. Defining the Issue Table (Junction Table)
CREATE TABLE issue (
    -- These must match parent domains exactly for [Referential Integrity]
    roll_no MEDIUMINT UNSIGNED NOT NULL,
    isbn CHAR(13) CHARACTER SET ascii NOT NULL,
    doi DATE NOT NULL,
    
    -- [Key Constraint]: Composite Primary Key
    -- This ensures a student cannot issue the same book on the same day twice.
    PRIMARY KEY(roll_no, isbn, doi),
    
    -- [Referential Integrity Constraints]: 
    -- Ensures 'roll_no' and 'isbn' MUST exist in the parent tables.
    FOREIGN KEY (roll_no) REFERENCES student(roll_no),
    FOREIGN KEY (isbn) REFERENCES book(isbn)
);
